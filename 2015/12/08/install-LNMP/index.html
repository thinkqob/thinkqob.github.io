install LNMP

Posted on 2015-12-08   |   In Python tutorial

[root@loaclhost ~]$ sudo vi /etc/sudoers user ALL=(ALL) ALL

[user@loaclhost ~]$ sudo yum update
[user@loaclhost ~]$ sudo yum upgrade

安装nginx
[user@loaclhost ~]$ sudo yum install epel-release
[user@loaclhost ~]$ sudo yum install nginx
[user@loaclhost ~]$ sudo systemctl start nginx
[user@loaclhost ~]$ sudo systemctl enable nginx.service
[user@loaclhost ~]$ sudo systemctl reload nginx.service

[user@loaclhost ~]$ sudo vi /etc/nginx/conf.d/user.conf

server {
listen 80;
server_name domainname.com;

# note that these lines are originally from the "location /" block
root   /home/user/www;
index index.php index.html index.htm;

location / {
    try_files $uri $uri/ =404;
}
error_page 404 /404.html;
error_page 500 502 503 504 /50x.html;
location = /50x.html {
    root /usr/share/nginx/html;
}

location ~ \.php$ {
    try_files $uri =404;
    fastcgi_split_path_info ^(.+\.php)(/.+)$;
    fastcgi_pass unix:/var/run/php-fpm/php-fpm.sock;
    fastcgi_index index.php;
    fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
    include fastcgi_params;
}

}

安装php
[user@loaclhost ~]$ sudo yum install php php-mysql php-fpm
[user@loaclhost ~]$ sudo systemctl start php-fpm
[user@loaclhost ~]$ sudo systemctl enable php-fpm
[user@loaclhost ~]$ sudo vi /etc/php-fpm.d/www.conf
listen = /var/run/php-fpm/php-fpm.sock
listen.owner = nobody
listen.group = nobody
user = nginx
group = nginx

新建用户
[user@loaclhost ~]$ sudo useradd -g nginx user2
[user@loaclhost ~]$ sudo chmod 750 /home/user2/www -R

安装mysql数据库
[user@loaclhost ~]$ sudo yum install mariadb mariadb-server

初始化设置mysql
[user@loaclhost ~]$ sudo mysql_secure_installation

用root账号登陆mysql
[user@loaclhost ~]$ sudo mysql -u root -p

创建数据库和用户
MariaDB [(none)]> CREATE DATABASE database_name;
MariaDB [(none)]> CREATE USER ‘user_name’@’localhost’ IDENTIFIED BY ‘password’;
MariaDB [(none)]> GRANT ALL PRIVILEGES ON database_name.* TO ‘user_name’@’localhost’;
MariaDB [(none)]> FLUSH PRIVILEGES;

查看所有用户
MariaDB [(none)]> MariaDB [(none)]> SELECT user,host,user FROM mysql.user;

参看所有数据库
MariaDB [(none)]> SHOW DATABASES;

参看所有表
MariaDB [(none)]> USE database_name;
MariaDB [database_name]> SHOW TABLES;

退出
MariaDB [(none)]> exit

安装python
[user@loaclhost ~]$ sudo yum install python python-devel python-setuptools python-pip
[user@loaclhost ~]$ sudo yum install python-lxml python-cffi
[user@loaclhost ~]$ sudo pip install –upgrade pip

安装scrapy
[user@loaclhost ~]$ sudo yum install libffi2 libffi-devel
[user@loaclhost ~]$ sudo yum install openssl openssl-devel
[user@loaclhost ~]$ sudo yum install libxml2 libxml2-devel
[user@loaclhost ~]$ sudo yum install gcc
[user@loaclhost ~]$ sudo pip install scrapy

安装flask
[user@loaclhost ~]$ sudo pip install flask
