Install SaltStack on CentOS 7

2016-08-23 | Categorized in SaltStack

[Minion:]
‘’’
[root@Minion ~]# sudo yum install https://repo.saltstack.com/yum/redhat/salt-repo-latest-1.el7.noarch.rpm
[root@Minion ~]# sudo yum clean expire-cache
[root@Minion ~]# sudo yum install salt-minion
[root@Minion ~]# sudo vi/etc/salt/minion

master: master
hash_type: sha256

[root@Master ~]# sudo systemctl start salt-master.service
[root@Master ~]# sudo systemctl enable salt-master.service
‘’’

[Master:]
‘’’
[root@Master ~]# sudo yum install https://repo.saltstack.com/yum/redhat/salt-repo-latest-1.el7.noarch.rpm
[root@Master ~]# sudo yum clean expire-cache
[root@Master ~]# sudo yum install salt-master

[root@Master ~]# sudo vi/etc/salt/master

interface: 0.0.0.0
hash_type: sha256

[root@Master ~]# sudo systemctl start salt-master.service
[root@Master ~]# sudo systemctl enable salt-master.service

[root@Master ~]# sudo firewall-cmd –get-active-zones
[root@Master ~]# sudo firewall-cmd –permanent –zone=public –add-port=4505-4506/tcp
[root@Master ~]# sudo firewall-cmd –reload
‘’’
‘’’
[root@Master ~]# salt-key -L
Accepted Keys:
Denied Keys:
Unaccepted Keys:
Minion
Rejected Keys:
[root@Master ~]# salt-key -a Minion
The following keys are going to be accepted:
Unaccepted Keys:
Minion
Proceed? [n/Y] y
[root@Master ~]# salt-key -L
Accepted Keys:
Minion
Denied Keys:
Unaccepted Keys:
Rejected Keys:
[root@Master ~]# salt Minion test.ping
Minion:
True
[root@Master ~]# salt Minion cmd.run “pwd”
Minion:
/root
‘’’

[else:]
‘’’
sudo yum install salt-master
sudo yum install salt-minion
sudo yum install salt-ssh
sudo yum install salt-syndic
sudo yum install salt-cloud
sudo yum install salt-api
salt-key -L # list all the keys
salt-key -a # accept key
salt-key -d # delete key
salt-key -A # accept all the keys
salt-key -D # delete all the keys
salt-key -R # reject all pending keys
‘’’
